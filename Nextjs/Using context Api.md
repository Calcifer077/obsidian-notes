You can use **context API** provided by react inside your next.js app. Steps are simple, just create a context like you usually do (it should be client component. `use client` at the top). Provide this context to our app by passing it in the root layout. Keep in mind that only **client components** can access it.

Creating a context API:
[Context API with `useReducer`](../React/Context%20API%20with%20`useReducer`.md)

Passing it to our app:
```tsx
import type { Metadata } from "next";
import "./globals.css";
import { IdeasContextProvider } from "./_components/IdeasContext";
import { convertMarkdown } from "./_lib/convertMarkdown";

export const metadata: Metadata = {
  title: "Create Next App",
  description: "Generated by create next app",
}; 

export default async function RootLayout({
  children,
}: Readonly<{
  children: React.ReactNode;
}>) {
  const { ideas } = await convertMarkdown();

  return (
    <html
      lang="en"
      suppressHydrationWarning
      className={`${inter.variable} h-full antialiased`}
    >
      <body
        suppressHydrationWarning
        className="h-screen w-screen flex flex-col bg-background text-foreground transition-colors"
      >
          <IdeasContextProvider initialIdeas={ideas}>
            {children}
          </IdeasContextProvider>
      </body>
    </html>
  );
}
```
